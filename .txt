import { Component, OnInit } from '@angular/core';
import { FormsModule } from '@angular/forms';
import { Router } from '@angular/router';
import { DriverService } from '../../../shared/services/driver.service';

declare var initSendOTP: any;
declare var sendOtp: any;
declare var verifyOtp: any;

@Component({
  selector: 'app-login',
  standalone: true,
  imports: [FormsModule],
  templateUrl: './login.component.html',
  styleUrl: './login.component.scss'
})
export class LoginComponent implements OnInit {

  phone = '';
  otp = '';
  step: 'phone' | 'otp' = 'phone';

  constructor(private router: Router, private driverService: DriverService) {}

  ngOnInit(): void {
    initSendOTP({
      widgetId: "356b736c7649373834323630",
      tokenAuth: "{token}",
      exposeMethods: true,

      success: (data: any) => {
        console.log("GLOBAL SUCCESS:", data);
      },

      failure: (error: any) => {
        console.log("GLOBAL FAILURE:", error);
      }
    });
  }

  async sendOtp() {
    if (!this.phone) return alert("Enter phone number");

    const formatted = "91" + this.phone;

    const exists = await this.driverService.driverExists(formatted);

    if (exists) {
      sendOtp(formatted,
        () => {
          console.log("OTP SENT SUCCESS");
          this.step = 'otp';
        },
        (err: any) => {
          console.error("OTP SEND FAILED", err);
          alert("OTP sending failed");
        }
      );
    } else {
      alert("User not found! Redirecting to signup...");
      this.router.navigate(['/signup'], {
        queryParams: { phone: this.phone }
      });
    }
  }

  async verifyOtpCode() {

    verifyOtp(this.otp,
      async () => {

        console.log("OTP Verified!");

        const driver = await this.driverService.getDriver("91" + this.phone);

        console.log("LOGIN SUCCESS", driver);

        this.router.navigate(['/driver/dashboard']);
      },
      () => {
        alert("Invalid OTP!");
      }
    );
  }
}
